import { Section, Preview, Snippet, Code } from '@wpmudev/storybook';
import { Button, LoadingButton } from '../../../src';
import dedent from 'dedent';

export const DisabledButtonCode = dedent`
import React from 'react';\n
import { Button } from '@wpmudev/sui-react';\n
const Example() => {
	return <Button disabled={true}>Disabled button</Button>;
}\n
export default Example;
`;

export const LoadingButtonCode = dedent`
import React from 'react';\n
import { LoadingButton } from '@wpmudev/sui-react';\n
const Example() => {
	return <LoadingButton isLoading>Loading button</Button>;
}\n
export default Example;
`;

<Section
	title="Disabled"
	contained={true}
	border={true}
	style={{ paddingRight: 0, paddingLeft: 0 }}
>
	<p>
		Set the <Code small={true}>isDisabled</Code> prop to disable a button
		that is not usable.
	</p>
	<Preview wrapper={true}>
		<Button disabled={true}>Disabled button</Button>
	</Preview>
	<Snippet language="js">{DisabledButtonCode}</Snippet>
</Section>

<Section
	title="Loading"
	contained={true}
	style={{ paddingRight: 0, paddingBottom: 0, paddingLeft: 0 }}
>
	<p>
		Set the <Code small={true}>isLoading</Code> prop to indicate the button
		is loading. The button text hides, and a spinner shows in its place.
	</p>
	<Preview wrapper={true}>
		<LoadingButton isLoading={true}>Loading button</LoadingButton>
	</Preview>
	<Snippet language="js">{LoadingButtonCode}</Snippet>
</Section>
